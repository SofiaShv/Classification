class CenteredFormatter(mpl_ticker.ScalarFormatter):
    """Acts exactly like the default Scalar Formatter, but yields an empty
    label for ticks at "center"."""
    center = 0
    def __call__(self, value, pos=None):
        if value == self.center:
            return ''
        else:
            return mpl_ticker.ScalarFormatter.__call__(self, value, pos)

def prep_graph():
    """Creates and configures a figure"""
    fig, ax = plt.subplots(figsize=(9, 9))
    ax.set_aspect('equal')

    # Center axis on the point (0, 0)
    ax.spines['left'].set_position(('data', 0))
    ax.spines['bottom'].set_position(('data', 0))

    # Eliminate upper and right axes
    ax.spines['right'].set_color('none')
    ax.spines['top'].set_color('none')

    # Show ticks in the left and lower axes only
    ax.xaxis.set_ticks_position('bottom')
    ax.yaxis.set_ticks_position('left')

    for axis, center in zip([ax.xaxis, ax.yaxis], [0, 0]):
        # Turn on minor and major gridlines and ticks
        axis.grid(True, 'major', ls='solid', lw=0.3, color='gray')
        axis.grid(True, 'minor', ls='solid', lw=0.1, color='gray')
        axis.set_minor_locator(mpl_ticker.AutoMinorLocator())

        # Hide the ticklabels at <0, 0>
        formatter = CenteredFormatter()
        formatter.center = center
        axis.set_major_formatter(formatter)
        
    # force always drawing the origin (0,0)
    ax.scatter([0], [0], c='red', alpha=0)
    
    return fig, ax

def plot_points(ax, ps, labels=None, c=None):
    points_layer = 100 # above all (but labels)
    labels_layer = points_layer + 1
    xs, ys = zip(*ps) # convert list of pairs [(a,b),(c,d),(e,f)] into pair of lists ([a,c,e],[b,d,f])

    scatter = ax.scatter(xs, ys, c=c, zorder=points_layer)

    if len(ps) < 20:
        # add coordinate labels
        for x, y in ps:
            # text of the label: coordinates '(x,y)' rounded to 0.01
            t = "({:.2g}, {:.2g})".format(x, y)
            ax.annotate(t,
                xy=(x, y), xytext=(-6, 6), # above and to the left of the point
                textcoords='offset points', # xytext is an offest to xy in points (not pixels)
                ha='right', va='bottom', # text is aligned to the right and bottom of its box
                size='x-small', # font size
                color='grey', # text color
                bbox=dict(ec='none', fc='none'), # do not draw a box around the label
                zorder=labels_layer)
    
    if labels and len(labels) == len(ps):
        # add custom labels
        for i, (x, y) in enumerate(ps):
            ax.annotate(labels[i],
                xy=(x, y), xytext=(20, 6), # above and to the right of the point
                textcoords='offset points', # xytext is an offest to xy in points (not pixels)
                ha='right', va='bottom', # text is aligned to the right and bottom of its box
                size='x-small', # font size
                color='grey', # text color
                bbox=dict(boxstyle='round', ec='grey', fc='white', alpha=0.7),
                family='monospace',
                zorder=labels_layer)

    return scatter

def max_points (points):
    leng=0
    for i in range(len(points)):
        ai=points[i]
        for j in range(i+1,len(points)):
            aj=points[j]
            if point_dist (ai, aj) > leng :
                leng = point_dist (ai, aj)
                max_point1 = ai
                max_point2 = aj

    if max_point1 [1] < max_point2 [1] :
        c = max_point2
        max_point2 =  max_point1
        max_point1 =  c
        
    return (max_point1, max_point2)

def farthest_points (points, A, B, C):
    leng = 0 
    max_leng=0
    min_leng=0 
    for p in points:
        leng = point_line_dist (p, A, B, C)
        if (leng >= max_leng):
            max_leng = leng 
            first_point = p

        if (leng <= min_leng):
            min_leng = leng 
            second_point = p

    return (first_point, second_point)

def proection (p, A, B, C):
    pr =[0,0]
    pr [0] = (B *(B*p[0]-A*p[1])-A*C) / ( A**2 + B**2)
    pr [1] = (A *(-B*p[0]+A*p[1])-B*C) / ( A**2 + B**2)
    return pr

def find_rectangle (max_point1, max_point2, first_point, second_point, A, B, C):

    first_proection = proection ( first_point, A, B, C)

    right_down = [0,0]

    right_down [0] = max_point1[0] - first_proection[0] + first_point[0]
    right_down [1] = max_point1[1] - first_proection[1] + first_point[1]

    left_down = [0,0]

    left_down [0] = max_point2[0] - first_proection[0] + first_point[0]
    left_down [1] = max_point2[1] - first_proection[1] + first_point[1]

    second_proection = proection ( second_point, A, B, C)

    right_up = [0,0]
    left_up = [0,0]

    right_up [0] = max_point1[0] - second_proection[0] + second_point[0]
    right_up [1] = max_point1[1] - second_proection[1] + second_point[1]

    left_up [0] = max_point2[0] - second_proection[0] + second_point[0]
    left_up [1] = max_point2[1] - second_proection[1] + second_point[1]
    
    return ( left_up, left_down, right_up, right_down )

def rotate (ps, k):
    c = []
    cos_k = cos (k)
    sin_k = sin (k)
    for x,y in ps:
        c1 = x 
        x = cos_k * x + sin_k * y
        y = -sin_k * c1 + cos_k * y
        c.append([x,y])
    return c

def ellipse_petunina(a):
    max_point1, max_point2 = max_points (a)
    
    A= max_point1[1] - max_point2[1]
    B= max_point2[0] - max_point1[0]
    C= max_point1[0] * max_point2[1] - max_point2[0] * max_point1[1]
    
    first_point, second_point = farthest_points (a, A, B, C)
    
    origin_rect = find_rectangle (max_point1, max_point2, first_point, second_point, A, B, C)
    
    k = atan (- A / B )
    
    rect = rotate (origin_rect, k)
    points = rotate (a, k)
    
    new_rect = centr (rect, rect[1])
    new_points = centr (points, rect[1])
    
    ru = new_rect [2]
    A2 = ru [1]
    B2 = ru [0]
    coef = A2/B2
    
    brand_new_rect = square (new_rect, coef)
    brand_new_points = square (new_points, coef)
    
    brand_new_center = find_center(brand_new_rect)
    r = find_radius(brand_new_points, brand_new_center)
    
    el_width = 2*r / coef
    el_height = 2*r
    [new_center] = square([brand_new_center], 1/coef)
    
    shift_x, shift_y = rect[1]
    [center] = centr([new_center], [-shift_x, -shift_y])
    
    [origin_center] = rotate([center], -k)
    
    return (origin_center, el_width, el_height, k)

def mvee(points, tol = 0.001):
    """
    Find the minimum volume ellipse.
    The solver is based on Khachiyan Algorithm, and the final solution is
    different from the optimal value by the pre-specified amount of 'tolerance'.

    Return A, c where the equation for the ellipse given in "center form" is
    (x-c).T * A * (x-c) = 1
    """
    points = np.asmatrix(points)
    N, d = points.shape
    Q = np.column_stack((points, np.ones(N))).T
    err = tol+1.0
    u = np.ones(N)/N
    while err > tol:
        # assert u.sum() == 1 # invariant
        X = Q * np.diag(u) * Q.T
        M = np.diag(Q.T * la.inv(X) * Q)
        jdx = np.argmax(M)
        step_size = (M[jdx]-d-1.0)/((d+1)*(M[jdx]-1.0))
        new_u = (1-step_size)*u
        new_u[jdx] += step_size
        err = la.norm(new_u-u)
        u = new_u
    c = u*points
    A = la.inv(points.T*np.diag(u)*points - c.T*c)/d    
    return np.asarray(A), np.squeeze(np.asarray(c))

def ellipse_other(points):
    cov_matrix, centroid = mvee (points)

    U, D, eigen_vectors = la.svd(cov_matrix)
    height, width = [2/np.sqrt(d) for d in D]
    
    # first eigen vector defines the major axis for the ellipse
    eiv1 = eigen_vectors[0]
    # tan(alpha) = eiv.y / eiv.x
    alpha = atan(eiv1[0] / eiv1[1])
    # but it may be directed 180 opposite to actual, so have to always flip
    alpha = abs(alpha)
    
    return (centroid, width, height, alpha)
