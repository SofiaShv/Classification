def make_vocabulary(docs):
    all_words = []
    for doc in docs:
        all_words += [word for word in doc.split() if word.isalpha() and len(word) > 1]
    popular_words = [word for word, _ in Counter(all_words).most_common(3000)]
    return set(popular_words)

def extract_features(vocab, docs): 
    doc_words = []
    for idx, doc in enumerate(docs):
        word_counts = dict((word, 0) for word in vocab)
        for word in doc.split():
            if word in vocab:
                word_counts[word] += 1
        doc_words.append(word_counts)
    return doc_words

# sample_classes: true clasess of each sample
def train(samples, sample_classes):
    sample_count = len(samples)
    
    # class -> number of samples
    class_counts = defaultdict(lambda: 0)
    # class, feature -> number of appearances of feature in samples of this class
    feature_counts = defaultdict(lambda: 0)
    # class -> a total number of features in this class
    feature_sums = defaultdict(lambda: 0)

    for sample_id, features in enumerate(samples):
        sample_class = sample_classes[sample_id]
        class_counts[sample_class] += 1
        for feature, feature_count in features.items():
            feature_counts[sample_class, feature] += feature_count
            # +1 is a part of Laplace smoothing
            feature_sums[sample_class] += (feature_count + 1)
    
    class_prob = dict()
    for klass, class_count in class_counts.items():
        class_prob[klass] = class_count / sample_count

    cond_prob = dict()
    for (klass, feature), feature_count in feature_counts.items():
        cond_prob[klass, feature] = (feature_count + 1) / feature_sums[klass]
    
    return class_prob, cond_prob

# only one sample at a time!
def estimate(classifier, features, klass):
    class_prob, cond_prob = classifier
    score = log(class_prob[klass])
    for feature, feature_count in features.items():
        score += log(cond_prob[klass, feature]) * feature_count
    return score

# only one sample at a time!
def classify(classifier, features):
    class_prob, _ = classifier
    
    class_scores = defaultdict(lambda: 0)
    for klass in class_prob.keys():
        class_scores[klass] = estimate(classifier, features, klass)
    
    return max(class_prob.keys(), key = lambda klass: class_scores[klass])

def classify_all(classifier, samples):
    return [classify(classifier, features) for features in samples]

def plot_roc_curve(classifier, samples, sample_labels):
    predicted_scores = [estimate(classifier, sample, 0) / estimate(classifier, sample, 1) for sample in test_samples]
    max_score = max(predicted_scores)
    predicted_scores = [score / max_score for score in predicted_scores]
    
    #  
    false_positive_rates = [] # 1 - 
    true_positive_rates = [] # 

    for threshold in range(0, 1000):
        fp, fn, tp, tn = (0, 0, 0, 0)
        
        for sample_id, spam_prob in enumerate(predicted_scores):
            if spam_prob > (threshold / 1000):
                if sample_labels[sample_id] == 1:
                    tp += 1
                else:
                    fp += 1
            else:
                if sample_labels[sample_id] == 1:
                    fn += 1
                else:
                    tn += 1

        sensitivity = tp / (tp + fn)
        specificity = tn / (tn + fp)
        false_positive_rates.append(1 - specificity)
        true_positive_rates.append(sensitivity)
    
    #  
    fig = Figure()
    FigureCanvas(fig)

    ax = fig.add_subplot(111)
    ax.set_title('Receiver Operating Characteristic')
    ax.set_xlabel('False Positive Rate (1 - specificity)')
    ax.set_xlim([-0.05, 1.05])
    ax.set_ylabel('True Positive Rate (sensitivity)')
    ax.set_ylim([-0.05, 1.05])
    ax.plot([0, 1], [0, 1], '--', linewidth=1, c='red', alpha=0.5) # the coin-toss line

    ax.plot(false_positive_rates, true_positive_rates, c='dodgerblue')

    return fig
