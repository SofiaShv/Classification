r = Random(0xdead)

def rnd (m, i):
    j = r.randint(0, m - 1) # randint's range is inclusive on both sides
    while j == i:
        j = r.randint(0, m - 1)
    return j

def E (train_set, train_set_Y, alpha, b, x ):
    E1 = 0

    for i, x_i in enumerate (train_set): 
        E1 = E1 + alpha[i] * train_set_Y[i] * np.inner(x_i, x)

    return E1 + b

def B (train_set, Y, alpha, C, b, i, j, E_i, E_j, alpha_old_i, alpha_old_j):
    def b1():
        return b - E_i - \
            Y[i] * (alpha[i] - alpha_old_i) * np.inner(train_set[i], train_set[i]) - \
            Y[j] * (alpha[j] - alpha_old_j) * np.inner(train_set[i], train_set[j])

    def b2():
        return b - E_j - \
            Y[i] * (alpha[i] - alpha_old_i) * np.inner(train_set[i], train_set[j]) - \
            Y[j] * (alpha[j] - alpha_old_j) * np.inner(train_set[j], train_set[j])

    if 0 < alpha[i] < C:
        return b1()

    if 0 < alpha[j] < C:
        return b2()

    return (b1() + b2()) / 2

def calc_alpha_j(train_set, alpha, Y, i, j, E_i, E_j, C):
    def LH():
        if Y[i] != Y[j] :
            L = max (0, alpha[j] - alpha[i])
            H = min (C, C + alpha[j] - alpha[i])
        else:
            L = max (0, alpha[j] + alpha[i] - C)
            H = min (C, alpha[j] + alpha[i])

        return (L, H)

    def ETA():
        return 2 * np.inner (train_set [i], train_set [j]) - \
          np.inner (train_set [i], train_set [i]) - \
          np.inner (train_set [j], train_set [j])
    
    alpha_old_j = alpha[j]

    L, H = LH()
    if L == H:
        return False
    
    eta = ETA()
    if eta >= 0 :
        return False
    
    alpha[j] = alpha[j] - Y[j] * (E_i - E_j) / eta

    if alpha[j] > H:
        alpha[j] = H
    elif alpha[j] < L:
        alpha[j] = L
    
    if abs(alpha[j] - alpha_old_j) < 10**(-5) :
        return False
    
    return True

def train(train_set, Y, C = 10, max_passes = 100, tol = 10**-5):
    m = len(train_set)
    
    alpha = [0 for i in range(m)]
    b = 0
    
    passes = 0
    while (passes < max_passes):
        num_changed_alphas = 0
        for i in range(m):
            E_i = E(train_set, Y, alpha, b, train_set[i]) - Y[i]
            e = Y[i] * E_i

            if ( e < -tol and alpha[i] < C ) or ( e > tol and alpha[i] > 0 ):
                j = rnd(m, i)

                E_j = E(train_set, Y, alpha, b, train_set[j]) - Y[j]

                alpha_old_i = alpha[i]
                alpha_old_j = alpha[j]
                
                if not calc_alpha_j(train_set, alpha, Y, i, j, E_i, E_j, C):
                    continue

                alpha[i] = alpha[i] + Y[i] * Y[j] * (alpha_old_j - alpha[j]) 

                #threshold
                b = B(train_set, Y, alpha, C, b, i, j, E_i, E_j, alpha_old_i, alpha_old_j)

                num_changed_alphas = num_changed_alphas + 1

        if (num_changed_alphas == 0):
            passes = passes + 1
        else:
            passes = 0
    
    return (train_set, Y, alpha, b)

def select_set(groups, class1, class2):
    ts = groups[class1] + groups[class2]
    ts_y = [1] * len(groups[class1]) + [-1] * len(groups[class2])
    return (ts, ts_y)

def eval_classifier(class1, class2, classifier):
    true_positives = 0
    true_negatives = 0
    false_positives = 0
    false_negatives = 0
    
    
    train_set, train_set_Y, alpha, b = classifier 

    for res in class1:
        
        E_x = E (train_set, train_set_Y, alpha, b, res )
        
        if E_x > 0:
            true_positives += 1
        else:
            false_negatives += 1
    
    for res in class2:
        
        E_x = E (train_set, train_set_Y, alpha, b,res )
        
        if E_x > 0:
            
            false_positives += 1
        else:
            true_negatives += 1


    #  (true positive rate, tpr)
    sensitivity = true_positives / (true_positives + false_negatives)

    #  (true negative rate, tnr)
    specificity = true_negatives / (true_negatives + false_positives)
    
    return dict(sensitivity=sensitivity, specificity=specificity)

def eval_normalized_classifier(class1_eval, class2_eval, threshold):
    true_positives = 0
    true_negatives = 0
    false_positives = 0
    false_negatives = 0
    

    for E_x in class1_eval:
        
        if E_x > threshold:
            true_positives += 1
        else:
            false_negatives += 1
    
    for E_x in class2_eval:
        
        if E_x > threshold:
            
            false_positives += 1
        else:
            true_negatives += 1


    #  (true positive rate, tpr)
    sensitivity = true_positives / (true_positives + false_negatives)

    #  (true negative rate, tnr)
    specificity = true_negatives / (true_negatives + false_positives)
    
    return dict(sensitivity=sensitivity, specificity=specificity)

def plot_roc_curve( class1_set, class2_set, classifier):
    #  
    false_positive_rates = [] # 1 - 
    true_positive_rates = [] # 
    
    train_set, train_set_Y, alpha, b = classifier
    
    
    # find the min and max
    estimates_raw = [ ]
    for ir in class1_set + class2_set:
        estimates_raw.append( E(train_set, train_set_Y, alpha, b, ir ) )
    estimate_min = min(estimates_raw)
    estimate_max = abs(estimate_min) + max(estimates_raw)
    
    # calculate estimates
    class_1_estimates = [ (E(train_set, train_set_Y, alpha, b, ir ) + abs(estimate_min))/estimate_max for ir in class1_set ]
    class_2_estimates = [ (E(train_set, train_set_Y, alpha, b, ir ) + abs(estimate_min))/estimate_max for ir in class2_set ]
    

    for threshold in range(0, 101, 1):
       
        classifier_quality = eval_normalized_classifier(class_1_estimates, class_2_estimates, threshold/100)

        fpr = 1 - classifier_quality['specificity']
        false_positive_rates.append(fpr)

        tpr = classifier_quality['sensitivity']
        true_positive_rates.append(tpr)
    
    #  
    fig = Figure()
    FigureCanvas(fig)

    ax = fig.add_subplot(111)
    ax.set_title('Receiver Operating Characteristic')
    ax.set_xlabel('False Positive Rate (1 - specificity)')
    ax.set_xlim([-0.05, 1.05])
    ax.set_ylabel('True Positive Rate (sensitivity)')
    ax.set_ylim([-0.05, 1.05])
    ax.plot([0, 1], [0, 1], '--', linewidth=1, c='red', alpha=0.5) # the coin-toss line

    ax.plot(false_positive_rates, true_positive_rates, c='dodgerblue')
    ax.plot(false_positive_rates, true_positive_rates, 'o', c='dodgerblue', alpha=0.4)

    return fig
