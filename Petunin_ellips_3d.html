from matplotlib.text import Annotation

class Annotation3D(Annotation):
    '''Annotate the point xyz with text s'''

    def __init__(self, s, xyz, *args, **kwargs):
        Annotation.__init__(self,s, xy=(0,0), *args, **kwargs)
        self._verts3d = xyz        

    def draw(self, renderer):
        xs3d, ys3d, zs3d = self._verts3d
        xs, ys, zs = proj_transform(xs3d, ys3d, zs3d, renderer.M)
        self.xy=(xs,ys)
        Annotation.draw(self, renderer)

def annotate3D(ax, s, *args, **kwargs):
    '''add anotation text s to to Axes3d ax'''

    tag = Annotation3D(s, *args, **kwargs)
    ax.add_artist(tag)

def make_3d_plot():
    fig = plt.figure(figsize=(7, 7))
    ax = fig.add_subplot(111,  projection='3d')
    ax.set_aspect('equal')
    
    # add axes labels
    for axis in 'xyz':
        getattr(ax, 'set_{}label'.format(axis))(axis.upper())

    return fig, ax

def scale_3d_plot(ax):
    # from https://stackoverflow.com/a/19933125
    scaling = np.array([getattr(ax, 'get_{}lim'.format(dim))() for dim in 'xyz'])
    ax.auto_scale_xyz(*[[np.min(scaling), np.max(scaling)]] * 3)

def point_label(point):
    rounded_point = np.copy(point)
    # set elements those absolute values are close to 0.0
    # (within some tolerance, e.g. -8.5373e-16)
    # to be anctual 0.0
    rounded_point[abs(rounded_point) < 10e-12] = 0
    
    return "({:.2g}, {:.2g}, {:.2g})".format(*rounded_point)
    
def plot_points(points, fig=None, ax=None, c=None, add_labels=True):
    if not ax:
        fig, ax = make_3d_plot()
    
    xs, ys, zs = zip(*points)
    ax.scatter(xs, ys, zs, color=c)
    
    if add_labels and len(points) <= 20:
        for i, p in enumerate(points): 
            annotate3D(ax, s=point_label(p), xyz=p,
                   size='x-small', color='grey', xytext=(-3,3),
                   textcoords='offset points', ha='right', va='bottom')
    
    return fig, ax

def plot_axes(fig, ax):
    fig.tight_layout() # remove white margins
    scale_3d_plot(ax)

    origin = np.zeros(3)
    i_hat = np.array([1, 0, 0])
    j_hat = np.array([0, 1, 0])
    k_hat = np.array([0, 0, 1])
    min_x, max_x = ax.get_xlim()
    for basis, label in zip([i_hat, j_hat, k_hat], ['X', 'Y', 'Z']):
        axis = basis * max_x
        # draw an axis line
        ax.plot(*zip(origin, axis), color='C3', lw=1)
        # annotate axis
        annotate3D(ax, s=label, xyz=axis,
                   size='small', color='grey', xytext=(-3,3),
                   textcoords='offset points', ha='right', va='bottom')
    return fig, ax

def max_points(points):
    max_dist = 0
    for i, p1 in enumerate(points):
        for p2 in points[i+1:]:
            dist = point_dist(p1, p2)
            if dist > max_dist:
                max_point1 = p1
                max_point2 = p2
                max_dist = dist

    if max_point1[1] > max_point2[1]:
        max_point2, max_point1 = max_point1, max_point2
    
    return max_point1, max_point2

def rot_mat_x(alpha):
    return np.array([
        [1, 0, 0],
        [0, math.cos(alpha), -math.sin(alpha)],
        [0, math.sin(alpha), math.cos(alpha)]
    ])

def rot_mat_y(alpha):
    return np.array([
        [math.cos(alpha), 0, math.sin(alpha)],
        [0, 1, 0],
        [-math.sin(alpha), 0, math.cos(alpha)]
    ])

def rot_mat_z(alpha):
    return np.array([
        [math.cos(alpha), -math.sin(alpha), 0],
        [math.sin(alpha), math.cos(alpha), 0],
        [0, 0, 1]
    ])

def vec_len(v):
    return np.sqrt(np.sum(v**2))

def vec_angle(v1, v2):
    sign = 1 if np.dot(v1, v2) > 0 else -1
    return sign * math.acos(np.dot(v1, v2) / (vec_len(v1) * vec_len(v2)))

def vec_angle_plane(dir_v, plane_norm):
    # angle between line (defined by its direction vector |dir_v|)
    # and the plane (defined by its normal vector |plane_norm|)
    # according to http://www.cleverstudents.ru/line_and_plane/angle_between_line_and_plane.html
    return math.asin(abs(np.dot(dir_v, plane_norm)) / (vec_len(dir_v) * vec_len(plane_norm)))

def find_pped(points):
    bottom_left = np.zeros(3)
    top_right = np.zeros(3)
    for i in range(3):
        bottom_left[i] = np.min(points[:,i])
        top_right[i] = np.max(points[:,i])
    
    return bottom_left, top_right

def plot_ellipsoid(center, rx, ry, rz, rot, fig=None, ax=None, c=None):
    if not ax:
        fig, ax = make_3d_plot()

    u = np.linspace(0, 2 * np.pi, 100)
    v = np.linspace(0, np.pi, 100)

    xs = rx * np.outer(np.cos(u), np.sin(v))
    ys = ry * np.outer(np.sin(u), np.sin(v))
    zs = rz * np.outer(np.ones_like(u), np.cos(v))
    
    for idx in range(xs.shape[0]):
        for idy in range(ys.shape[1]):
            xs[idx,idy],ys[idx,idy],zs[idx,idy] = np.inner(rot, [xs[idx,idy],ys[idx,idy],zs[idx,idy]]) + center
    
    if not c:
        c = 'silver'

    # ax.plot_surface(xs, ys, zs, cstride=4, rstride=4, color='white', edgecolor='silver', alpha=0.4)
    ax.plot_wireframe(xs, ys, zs, cstride=4, rstride=4, color=c, alpha=0.6)

    return fig, ax

def ellipsoid_petunina(ps):
    origin = np.zeros(3)
    i_hat = np.array([1, 0, 0])
    j_hat = np.array([0, 1, 0])
    k_hat = np.array([0, 0, 1])

    diag = max_points(ps)
    
    shifted_ps = translate(ps, diag[0])
    shifted_diag = translate(diag, diag[0])
    
    proj_xz = np.copy(shifted_diag[1])
    proj_xz[1] = 0 # set y = 0, puts proj in plane XZ
    alpha = vec_angle(proj_xz, i_hat) # angle between proj and i
    print("alpha: {:.2f}".format(math.degrees(alpha)))
    rot_y = rot_mat_y(alpha) # find a rotation matrix for angle alpha around Y axis
    
    rot_xy_diag = rotate(shifted_diag, rot_y)
    
    psi = vec_angle(rot_xy_diag[1], i_hat) # angle between (now rotated) diagonal and i
    print("psi: {:.2f}".format(math.degrees(psi)))
    rot_z = rot_mat_z(-psi) # find a rotation matrix for angle psi around Z axis
    
    rot_matrix = np.dot(rot_z, rot_y)

    rotated_ps = rotate(shifted_ps, rot_matrix)
    rotated_diag = rotate(shifted_diag, rot_matrix)
    
    pped = find_pped(rotated_ps)
    
    square_coef = 1 / (pped[1] - pped[0])
    squaring_matrix = np.zeros((3,3))
    for i, coef in enumerate(square_coef):
        squaring_matrix[i,i] = coef
    
    square_ps = np.array([np.inner(squaring_matrix, p) for p in rotated_ps])
    square_diag = np.array([np.inner(squaring_matrix, p) for p in rotated_diag])
    square = np.array([np.inner(squaring_matrix, p) for p in pped])
    
    square_center = (square[0] + square[1]) / 2
    
    R = max([
        point_dist(square_center, p)
        for p in square_ps
    ])
    
    unsquaring_matrix = np.linalg.inv(squaring_matrix)
    pped_center = np.inner(unsquaring_matrix, square_center)
    a, b, c = unsquaring_matrix.diagonal() * R
    
    unrot_matrix = np.linalg.inv(rot_matrix)
    [shifted_center] = rotate([pped_center], unrot_matrix)

    [ellipsoid_center] = translate([shifted_center], -diag[0])
    
    return ellipsoid_center, a, b, c, unrot_matrix

def mvee(points, tol = 0.001):
    """
    Find the minimum volume ellipse.
    The solver is based on Khachiyan Algorithm, and the final solution is
    different from the optimal value by the pre-specified amount of 'tolerance'.

    Return A, c where the equation for the ellipse given in "center form" is
    (x-c).T * A * (x-c) = 1
    """
    points = np.asmatrix(points)
    N, d = points.shape
    Q = np.column_stack((points, np.ones(N))).T
    err = tol+1.0
    u = np.ones(N)/N
    while err > tol:
        # assert u.sum() == 1 # invariant
        X = Q * np.diag(u) * Q.T
        M = np.diag(Q.T * np.linalg.inv(X) * Q)
        jdx = np.argmax(M)
        step_size = (M[jdx]-d-1.0)/((d+1)*(M[jdx]-1.0))
        new_u = (1-step_size)*u
        new_u[jdx] += step_size
        err = np.linalg.norm(new_u-u)
        u = new_u
    c = u*points
    A = np.linalg.inv(points.T*np.diag(u)*points - c.T*c)/d    
    return np.asarray(A), np.squeeze(np.asarray(c))

def ellipsoid_other(points):
    cov_matrix, centroid = mvee(points)

    U, s, rotation = np.linalg.svd(cov_matrix)
    a,b,c = 1.0/np.sqrt(s)
    
    return (centroid, a, b, c, rotation.T)
